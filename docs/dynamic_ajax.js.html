/*! dynamic_ajax.js */

/**
 * This mixin provides a dynamic ajax driver dispatching mechanism.
 *
 * @class dynamic_ajax
 * @mixin
 */

var dynamic_mixin = require('../mixin/dynamic_mixin');

module.exports = function() {
  return mixin;

  function mixin() {
    return dynamic_mixin.call(this, 'ajax');
  }
};
